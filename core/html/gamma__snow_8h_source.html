186  inline double calc_df(const double a, const double b, const double z) const {
187  return a*b*exp(-(a + 1.0)*log(b) - lgamma(a + 1.0) + a*log(z) - z/b)
188  + 1.0 - z*exp(-a*log(b) - lgamma(a) + (a - 1.0)*log(z) - z/b)
189  - gamma_p(a, z/b);
190  }

192  inline double calc_q(const double a, const double b, const double z) const
193  {
194  return a*b*gamma_p(a + 1.0, z/b) + z*(1.0 - gamma_p(a, z/b));
195  }

197  double corr_lwc(const double z1, const double a1, const double b1,
198  double z2, const double a2, const double b2) const {
199  double zmax = z1;
200  double zmin = 0.0;
201  double dz = fabs(zmax - zmin);
202  double dz_old = dz;
203  const double rtol = 1.0e-4;
204 
205  const double Q1 = calc_q(a1, b1, z1);
206  const double precision = Q1*rtol;
207  double Q2 = calc_q(a2, b2, z2);
208  double f = Q2 - Q1;
209 
210  double df = calc_df(a2, b2, z2);
211 
212  size_t i;
213  size_t max_iter = 30;
214  for (i=0; i < max_iter; ++i) {
215  if (((z2 - zmax)*df - f)*((z2 - zmin)*df - f) > 0.0 || fabs(2.0*f) > fabs(dz_old*df)) {
216  dz_old = dz;
217  dz = (zmax - zmin)*0.5;
218  z2 = zmin + dz;
219  } else {
220  dz_old = dz;
221  dz = f/df;
222  z2 -= dz;
223  }
224  if (fabs(dz) < precision)
225  break;
226  // Prepare next iteration
227  Q2 = calc_q(a2, b2, z2);
228  f = Q2 - Q1;
229  df = calc_df(a2, b2, z2);
230  if (f < 0.0) zmin = z2;
231  else zmax = z2;
232  }
233  #ifdef __UNIT_TEST__
234  #if VERBOSE>0
235  std::cout << "Num iter in corr_lwc = " << i << std::endl;
236  #endif
237  #endif
238  if (i < max_iter) return z2;
239  else throw std::runtime_error("corr_lwc did not converge"); // No convergence in max_iter iterations!
240  }

242  void calc_snow_state(const double shape, const double scale, const double y0, const double lambda,
243  const double lwd, const double max_water_frac, const double temp_swe,
244  double& swe, double& sca) const {
245  double y = 0.0;
246  double y1 = 0.0;
247  const double m = shape*scale;
248  if (lambda <= 0.0) { // accumulated melt is -1.
249  swe = m;
250  sca = 1.0 - y0;
251  } else if (lambda/scale > 1.3*shape + 20.0) {
252  swe = sca = 0.0;
253  return;
254  } else {
255  const double x = lambda/scale;
256  y = gamma_p(shape, x);
257  y1 = y - exp(shape*log(x) - x - lgamma(shape))/shape;
258  swe = m*(1.0 - y1) - lambda*(1 - y);
259  sca = (1.0 - y)*(1.0 - y0);
260  }
261  if (lwd > m) swe *= 1.0 + max_water_frac;
262  else if (lwd > 0.0) {
263  const double sat = lwd/max_water_frac;
264  const double x = sat/scale;
265  const double ssa = gamma_p(shape, x);
266  const double ssa1 = ssa - exp(shape*log(x) - x - lgamma(shape))/shape;
267  const double liqwat = max_water_frac*(m*(ssa1 - y1) + sat*(1.0 - ssa) - lambda*(1.0 - y));
268  swe += liqwat;
269  }
270  swe += temp_swe;
271  swe *= 1.0 - y0;
272  }

289  /*

291  * updates the state and response upon return.

294  * \param T temperature degC, considered constant over timestep dt

305  // Some special cases treated first for efficiency
306 
307  // Read state vars needed early (possible early return)
308  double sdc_melt_mean = s.sdc_melt_mean;
309  double acc_melt = s.acc_melt;

314  acc_melt = iso_pot_energy = 0.0;
315 
316 
317  double snow;
318  double rain;

320  else {snow = 0.0;rain = prec;}
321  if (fabs(snow + rain - prec) > 1.0e-8)
322  throw std::runtime_error("Mass balance violation!!!!");
323 
324  if (snow < gamma_snow::tol && sdc_melt_mean < gamma_snow::tol && acc_melt < 0.0) { // Early autumn scenario, no snow
325  // Set some of the states.
326  s.albedo = p.max_albedo;
327  s.surface_heat = 0.0;
328  s.iso_pot_energy = 0.0;
329  r.sca = 0.0;
330  r.storage = 0.0;
331  r.outflow = prec_mm_h;
332  return;
333  }

336  double albedo = s.albedo;
337  double lwc = s.lwc;
338  double surface_heat = s.surface_heat;
339  double alpha = s.alpha;
340  double temp_swe = s.temp_swe;
341 
342  // Response vars;
343  double sca = 0.0;
344  double storage = 0.0;
345  double outflow = 0.0;
346 
347  // Local variables
348  const double min_albedo = p.min_albedo;
349  const double max_albedo = p.max_albedo;
350  const double snow_cv = p.effective_snow_cv(forest_fraction,altitude);
351  const double albedo_range = max_albedo - min_albedo;
352  const double dt_in_days = dt/double(calendar::DAY);
353  const double slow_albedo_decay_rate = 0.5*albedo_range*dt_in_days/p.slow_albedo_decay_rate;
354  const double fast_albedo_decay_rate = pow(2.0, -dt_in_days/p.fast_albedo_decay_rate);
355 
356 
357  const double T_k = T + 273.15; // Temperature in Kelvin
358  const double turb = p.wind_scale*wind_speed + p.wind_const;
359  double vapour_pressure = 33.864*(pow(7.38e-3*T + 0.8072, 8) - 1.9e-5*fabs(1.8*T + 48.0)
360  + 1.316e-3)*rel_hum;
361  if (T < 0.0) // Change VP over water to VP over ice (Bosen)
362  vapour_pressure *= 1.0 + 9.72e-3*T + 4.2e-5*T*T;
363 
364  if (snow > gamma_snow::tol)
365  albedo += snow*albedo_range/p.snowfall_reset_depth;
366  else {
367  if (T < 0.0)
368  albedo -= slow_albedo_decay_rate;
369  else
370  albedo = min_albedo + fast_albedo_decay_rate*(albedo - min_albedo);
371  }
372 
373  albedo = std::max(std::min(albedo, max_albedo), min_albedo);
374 
375  double effect = rad*(1.0 - albedo);
376  effect += 0.98*sigma*pow(vapour_pressure/T_k, 6.87e-2)*pow(T_k, 4);
377 
378  if (T > 0.0 && snow < gamma_snow::tol) // Why not if (rain > 0.0)?
379  effect += rain*T*water_heat/(double)dt;
380  if (T <= 0.0 && rain < gamma_snow::tol)
381  effect += snow*T*ice_heat/(double)dt;
382 
383  if (p.calculate_iso_pot_energy) {
384  double iso_effect = effect - BB0 + turb*(T + 1.7*(vapour_pressure - 6.12));
385  iso_pot_energy += iso_effect*(double)dt/melt_heat;
386  }
387 
388  double sst = std::min(0.0, 1.16*T - 2.09);
389  if (sst > -gamma_snow::tol)
390  effect += turb*(T + 1.7*(vapour_pressure - 6.12)) - BB0;
391  else
392  effect += turb*(T - sst + 1.7*(vapour_pressure - 6.132*exp(0.103*T - 0.186))) // Sensible heat contribution
393  - 0.98*sigma*pow(sst + 273.15, 4); // Upward longwave from colder snowpack.
394 
395  double delta_sh = -surface_heat; // Surface heat change, positive during warming
396  surface_heat = p.surface_magnitude*ice_heat*sst*0.5; // New surface heat; always nonpositive since sst <= 0.
397  delta_sh += surface_heat;
398 
399  double energy = effect*(double)dt;
400  if (delta_sh > 0.0) energy -= delta_sh; // Surface energy is a sink, but not a source
401 
402  double potential_melt = std::max(0.0, energy/melt_heat); // Potential snowmelt in mm.
403 
404  // Done surface layer energy and phase change calculations, now mass balance!
405  // The mass balance section is skipped off-season (when there is no snow pack).
406  // The distributed mass balance simulates the CV during the season. There is a
407  // static/parameter snowfall CV applied to new snow. If no mid-winter melt events
408  // occur, the melt-season SWE_CV will also have this value. If a melt event occurs,
409  // the melt is fastest from the thinnest snow packs, and the CV is thus increased.
410  // This is accomplished by reducing the shape alpha, keeping the scale constant.
411  // Subsequent snowfalls will then adjust the SWE_CV back towards the snowfall CV.
412 
413  // Completely bare ground lead to exit only after acc-melt is computed, in order to
414  // provide lambda values to the updating routine, which estimates elevation gradients.
415  // With smaller evenly-distributed snow packs at low elevation, this gradient would
416  // discontinue increasing with time if low-altitude accumulation were stopped.
417 
418  double sdc_scale = sdc_melt_mean/alpha;
419

421  lwc, p.max_water, temp_swe, storage, sca);
422 
423  double start_storage_value = storage;
424 
425  if (acc_melt < 0.0) {
426  if (snow < gamma_snow::tol) snow = 0.0;
427  else {
428  double alpha_prev = alpha;
429  double sdc_scale_prev = sdc_scale;
430  double sdc_snow = snow/(1.0 - p.initial_bare_ground_fraction);
431 
432  alpha = (sdc_melt_mean*alpha + sdc_snow/(snow_cv*snow_cv))/(sdc_snow + sdc_melt_mean);
433 
434  sdc_melt_mean += sdc_snow;

444  }

447  if (sdc_melt_mean <= potential_melt) {
448  storage = 0.0;
449  reset_snow_pack(sca, lwc, alpha, sdc_melt_mean, acc_melt, temp_swe, storage, p);
450  sdc_scale = 0.0;
451  } else if (potential_melt > 0.0) {
452  sdc_melt_mean -= potential_melt;
453  lwc += potential_melt;
454  // Update alpha and scale. Alpha in interval [0.1, 1.0/(snow_cv*snow_cv)]
455  alpha = std::max(0.1, sdc_melt_mean/sdc_scale);
456  if (alpha > 1.0/(snow_cv*snow_cv))
457  alpha = 1.0/(snow_cv*snow_cv);
458  sdc_scale = sdc_melt_mean/alpha;
459  }
460 
461  } else { // Spring snow pack
462  temp_swe += snow/(1.0 - p.initial_bare_ground_fraction);
463  if (temp_swe > 0.0) {
464  double melt = std::min(temp_swe, potential_melt);
465  temp_swe -= melt;
466  potential_melt -= melt;
467  lwc += melt;
468  if (temp_swe < gamma_snow::tol) temp_swe = 0.0;
469  }
470  acc_melt += potential_melt;
471  lwc += rain + potential_melt;
472  if (!p.calculate_iso_pot_energy || p.is_snow_season(t)) {
473  if (storage < std::max(0.2, 2*temp_swe) || storage < 0.2*rain) {
474  storage += snow;

476  sdc_scale = sdc_melt_mean/alpha;
477  }
478  }
479  }
480  // Establish the snow pack state after this time step
481  calc_snow_state(alpha, sdc_scale, p.initial_bare_ground_fraction, acc_melt,
482  lwc, p.max_water, temp_swe, storage, sca);
483 
484  outflow = prec + start_storage_value - storage;
485 
486  if (outflow < 0.0) outflow = 0.0; // Tiny rounding errors may occur.
487

489  s.albedo = albedo;
490  s.lwc = lwc;
491  s.surface_heat = surface_heat;
492  s.alpha = alpha;
493  s.sdc_melt_mean = sdc_melt_mean;

496  s.temp_swe = temp_swe;
