126  const std::vector<double>& values,
127  point_interpretation_policy interpretation=POINT_INSTANT_VALUE) {
128  if (times.size() == values.size() + 1) {

130  new GenericTs<point_timeseries<point_timeaxis> >(point_timeseries<point_timeaxis>(
131  point_timeaxis(times), values, interpretation)));
132  } else if (times.size() == values.size()) {
133  auto tx(times);

136  new GenericTs<point_timeseries<point_timeaxis> >(point_timeseries<point_timeaxis>(
137  point_timeaxis(tx), values, interpretation)));
138  } else {
139  throw std::runtime_error("create_time_point_ts times and values arrays must have corresponding count");
