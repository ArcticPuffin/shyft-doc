751  double run(const vector<double>& rp) {
752  auto p = expand_p_vector(rp);// expand to full vector, then:
753  parameter_accessor.set(p); // Sets global parameters, all cells share a common pointer.
754  reset_states();
755  model.run_cells();
756  double goal_function_value = 0.0;// overall goal-function, intially zero
757  double scale_factor_sum = 0.0; // each target-spec have a weight, -use this to sum up weights
758  vector<pts_t> catchment_d;
759  vector<area_ts> catchment_sca, catchment_swe;// "catchment" level simulated discharge,sca,swe
760  for (const auto& t : targets) {
761  shyft::time_series::direct_accessor<pts_t, timeaxis_t> target_accessor(t.ts, t.ts.ta);
762  pts_t property_sum;
763  switch (t.catchment_property) {
764  case DISCHARGE:
765  property_sum = compute_discharge_sum(t, catchment_d);
766  break;
767  case SNOW_COVERED_AREA:
768  property_sum = compute_sca_sum(t, catchment_sca);
769  break;
770  case SNOW_WATER_EQUIVALENT:
771  property_sum = compute_swe_sum(t, catchment_swe);
772  break;
773  case ROUTED_DISCHARGE:
774  property_sum = *model.river_output_flow_m3s(t.river_id);
775  break;
776  case CELL_CHARGE:
777  property_sum = compute_charge_sum(t, catchment_d);
778  }
779  shyft::time_series::average_accessor<pts_t, timeaxis_t> property_sum_accessor(property_sum, t.ts.ta);
780  double partial_goal_function_value;
781  if (t.calc_mode == target_spec_calc_type::NASH_SUTCLIFFE) {
782  partial_goal_function_value = nash_sutcliffe_goal_function(target_accessor, property_sum_accessor);
783  } else if(t.calc_mode == target_spec_calc_type::KLING_GUPTA) {
784  // ref. KLING-GUPTA Journal of Hydrology 377 (2009) 80–91, page 83, formula (10):
785  // a=alpha, b=betha, q =sigma, u=my, s=simulated, o=observed
786  partial_goal_function_value /* EDs */ = kling_gupta_goal_function<dlib::running_scalar_covariance<double>>(target_accessor,
787  property_sum_accessor,
788  t.s_r,
789  t.s_a,
790  t.s_b);
791  } else {
792  partial_goal_function_value = abs_diff_sum_goal_function(target_accessor,property_sum_accessor);
793  }
794  if (isfinite(partial_goal_function_value)) {
795  scale_factor_sum += t.scale_factor;
796  goal_function_value += t.scale_factor*partial_goal_function_value;
797  } else if (print_progress_level > 0) {
798  cout << "warning: goal-function " << t.catchment_property << ": evaluated as nan" << endl;
799  }
800  }
801  goal_function_value /= scale_factor_sum;
802  parameters_trace.push_back(parameter_accessor);// save to the parameters_trace
803  goal_fn_trace.push_back(goal_function_value);//
804  if (print_progress_level > 0) {
805  cout << goal_function_value <<" : ParameterVector(";
806  for (size_t i = 0; i < parameter_accessor.size(); ++i) {
807  cout << parameter_accessor.get(i);
808  if (i < parameter_accessor.size() - 1) cout << ", ";
809  }
810  cout << ")" << endl;
811  }
812  return goal_function_value;
813  }
