28  static inline arma::mat make_covariance(int n_daily_observations,double covariance_init,double hourly_correlation) {
29  arma::mat P(n_daily_observations,n_daily_observations);
30  double dt_hours=24.0/n_daily_observations;
31  for(int i=0;i<n_daily_observations;++i) {
32  for(int j=0;j<n_daily_observations;++j) {
33  // the correlation is proportional to the daily-folded distance between i and j
34  // so the last time-step in a day is closest to the first hour of the next day etc.
35  // this is essential to the time-folding and 'solar' driven pattern of bias.
36  int ix_diff= std::min(abs(i-j),n_daily_observations-abs(i-j));
37  P.at(i,j) = covariance_init*pow(hourly_correlation,ix_diff*dt_hours);
38  }
39  }
40  return P;
41  }
