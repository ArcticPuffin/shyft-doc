35  static inline arma::mat make_covariance(int n_daily_observations,double covariance_init,double hourly_correlation) {
36  arma::mat P(n_daily_observations,n_daily_observations);
37  double dt_hours=24.0/n_daily_observations;
38  for(int i=0;i<n_daily_observations;++i) {
39  for(int j=0;j<n_daily_observations;++j) {
40  // the correlation is proportional to the daily-folded distance between i and j
41  // so the last time-step in a day is closest to the first hour of the next day etc.
42  // this is essential to the time-folding and 'solar' driven pattern of bias.
43  int ix_diff= std::min(abs(i-j),n_daily_observations-abs(i-j));
44  P.at(i,j) = covariance_init*pow(hourly_correlation,ix_diff*dt_hours);
45  }
46  }
47  return P;
48  }
