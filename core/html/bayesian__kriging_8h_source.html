91  template<class S, class D, class P> void build_covariance_matrices(S source_begin, S source_end,
92  D destination_begin, D destination_end,
93  const P& parameter, arma::mat& K, arma::mat& k)
94  {
95  const arma::uword n = (arma::uword)std::distance(source_begin, source_end);
96  const arma::uword m = (arma::uword)std::distance(destination_begin, destination_end);
97  arma::vec source_dists(n*(n-1)/2);
98  arma::vec K_nz(n*(n-1)/2);
99  arma::uword idx = 0;
100  for (auto i=source_begin; i != source_end; ++i)
101  for (auto j=std::next(i); j != source_end; ++j)
102  source_dists.at(idx++) = scaled_dist(i->mid_point(), j->mid_point(), parameter);
103 
104  cov(source_dists, K_nz, parameter);
105  idx = 0;
106  K.eye(n,n);
107  for (arma::uword i = 0; i<n; ++i)
108  for (arma::uword j = i + 1; j<n; ++j)
109  K.at(i, j) = K_nz.at(idx++);
110  K.diag() *= zero_dist_cov(parameter);
111  K = arma::symmatu(K);
112 
113  arma::mat k_dists(n,m);
114  arma::uword i = 0;
115 
116  for (auto s=source_begin; s != source_end; ++s) {
117  arma::uword j = 0;
118  for (auto d=destination_begin; d != destination_end; ++d)
119  k_dists.at(i, j++) = scaled_dist(s->mid_point(), d->mid_point(), parameter);
120  ++i;
121  }
122  cov(k_dists, k, parameter);
123  }

278  template<class TSA, class S, class D, class T, class P>
279  void btk_interpolation(S source_begin, S source_end,
280  D destination_begin, D destination_end,
281  const T& time_axis, const P& parameter)
282  {
283  // Armadillo's notion of submatrix views are used for slicing out portions of F, k, and K
284  // to minimize the number of matrix allocations.
285 
286  // Allocate matrices of known sizes:
287  arma::mat22 H, H_inv, G, G_inv, GH_inv;
288  arma::mat::fixed<2,1> E_beta_pri, E_beta_w_pri,/* E_beta_post,*/ beta_hat;
289  // These matrices sizes vary with the number valid sources and the number of destinations.
290  arma::mat K, k, F, f, K_inv, E_beta_w, omega, BM, T_obs, E_temp_post;
291 
292  // Prior data
293  E_beta_pri(0, 0) = 0.0; // Old code says this is ok. TODO: Check assumption.
294  arma::mat22 eye22 = arma::diagmat(arma::vec(2, arma::fill::ones));
295 
296  // Gather spatial data for all stations and destinations
297  utils::build_elevation_matrices(source_begin, source_end, destination_begin, destination_end, F, f);
298  utils::build_covariance_matrices(source_begin, source_end, destination_begin, destination_end, parameter, K, k);
299 
300  // Build full operators
301 
302  K_inv = K.i();
303  H_inv = F.t()*K_inv*F;
304  if (arma::rank(H_inv) == 1) {
305  throw std::runtime_error("The bayestian temperature kriging algorithm needs at least two sources at different heights.");
306  }
307  H = H_inv.i();
308  G_inv = H_inv;
309  G_inv.at(1, 1) += 1/(parameter.temperature_gradient_sd()*parameter.temperature_gradient_sd());
310  G = G_inv.i();
311  GH_inv = G*H_inv;
312  BM = (f - F.t()*K_inv*k).t()*(eye22 - GH_inv);
313  E_beta_w = H*F.t()*K_inv; // beta_est_weights
314  omega = k.t()*K_inv; // krig_weights
315  // Reduced matrices
316  arma::mat F_r, E_beta_w_r, omega_r, GH_inv_r, BM_r;
317 
318  // Matrix pointers used in the time loop
319  arma::mat *F_p=nullptr, *E_beta_w_p=nullptr, *omega_p=nullptr, *GH_inv_p=nullptr, *BM_p=nullptr;
320 
321  const size_t num_sources = std::distance(source_begin, source_end);
322  std::vector<TSA> source_accessors;
323  source_accessors.reserve(num_sources);
324  std::for_each(source_begin, source_end, [&] (const typename S::value_type& source)
325  { source_accessors.emplace_back(TSA(source.temperatures(), time_axis)); });
326  // Time step loop
327  const size_t num_timesteps = time_axis.size();
328  std::vector<double> temperatures;
329  temperatures.reserve(num_sources);
330 
331  std::vector<arma::uword> valid_inds, prev_valid_inds;
332  valid_inds.reserve(num_sources);
333  for (size_t t_step=0; t_step < num_timesteps; ++t_step) {
334  temperatures.clear();
335  prev_valid_inds = valid_inds;
336  valid_inds.clear();
337  size_t idx = 0;
338  for (auto& a: source_accessors) {
339  double v = a.value(t_step);
340  if (std::isfinite(v)) {
341  valid_inds.push_back((arma::uword)idx);
342  temperatures.push_back(v);
343  }
344  ++idx;
345  }
346  if (valid_inds != prev_valid_inds || valid_inds.size()==0) {
347  if (valid_inds.size() == 0) {
348  //std::cout << "period("<< t_step <<"| " << num_timesteps << ") = " << time_axis.period(t_step) << std::endl;
349  throw std::runtime_error(std::string("bayesian kriging temperature: No valid sources for time period, giving up.") + calendar().to_string(time_axis.period(t_step)));
350  }
351  if (valid_inds.size() == num_sources) {
352  // Use full operators
353  F_p = &F;
354  E_beta_w_p = &E_beta_w;
355  omega_p = &omega;
356  GH_inv_p = &GH_inv;
357  BM_p = &BM;
358  } else {
359  // Build new reduced operators from full operators
360  arma::uvec sub_idx(valid_inds);
361  F_r = F.rows(sub_idx);
362  K_inv = K.submat(sub_idx, sub_idx).i();
363  H_inv = F_r.t()*K_inv*F_r;
364  H = H_inv.i();
365  G_inv = H_inv;
366  G_inv(1, 1) += 1/(parameter.temperature_gradient_sd()*parameter.temperature_gradient_sd());
367  G = G_inv.i();
368  GH_inv_r = G*H_inv;
369  arma::mat k_red = k.rows(sub_idx);
370  BM_r = (f - F_r.t()*K_inv*k_red).t()*(eye22 - GH_inv_r);
371  E_beta_w_r = H*F_r.t()*K_inv; // beta_est_weights
372  omega_r = k_red.t()*K_inv; // krieg_weights
373  // Assign pointers
374  F_p = &F_r;
375  E_beta_w_p = &E_beta_w_r;
376  omega_p = &omega_r;
377  GH_inv_p = &GH_inv_r;
378  BM_p = &BM_r;
379  }
380  }
381 
382  // Build prior data for time step:
383  E_beta_pri(1, 0) = parameter.temperature_gradient(time_axis.period(t_step));
384  E_beta_w_pri = ((eye22 - *GH_inv_p)*E_beta_pri);
385 
386  // Fill T_obs with valid temperatures
387  T_obs.set_size((arma::uword)valid_inds.size(), 1);
388  std::copy(std::begin(temperatures), std::end(temperatures), T_obs.begin_col(0));
389  // Core computational work here:
390  beta_hat = (*E_beta_w_p)*T_obs;
391  arma::mat T_hat = f.t()*beta_hat + (*omega_p)*(T_obs - (*F_p)*beta_hat);
392  //E_beta_post = (*GH_inv_p)*beta_hat + E_beta_w_pri;
393  E_temp_post = arma::vec(T_hat - (*BM_p)*(beta_hat - E_beta_pri));
394 
395  arma::uword dist = 0;
396  for (D d=destination_begin; d != destination_end; ++d)
397  d->set_temperature(t_step, E_temp_post.at(dist++));
398 
399  }
400  }
