85  template<class S, class D, class P> void build_covariance_matrices(S source_begin, S source_end,
86  D destination_begin, D destination_end,
87  const P& parameter, arma::mat& K, arma::mat& k)
88  {
89  const arma::uword n = (arma::uword)std::distance(source_begin, source_end);
90  const arma::uword m = (arma::uword)std::distance(destination_begin, destination_end);
91  arma::vec source_dists(n*(n-1)/2);
92  arma::vec K_nz(n*(n-1)/2);
93  arma::uword idx = 0;
94  for (auto i=source_begin; i != source_end; ++i)
95  for (auto j=std::next(i); j != source_end; ++j)
96  source_dists.at(idx++) = scaled_dist(i->mid_point(), j->mid_point(), parameter);
97 
98  cov(source_dists, K_nz, parameter);
99  idx = 0;
100  K.eye(n,n);
101  for (arma::uword i = 0; i<n; ++i)
102  for (arma::uword j = i + 1; j<n; ++j)
103  K.at(i, j) = K_nz.at(idx++);
104  K.diag() *= zero_dist_cov(parameter);
105  K = arma::symmatu(K);
106 
107  arma::mat k_dists(n,m);
108  arma::uword i = 0;
109 
110  for (auto s=source_begin; s != source_end; ++s) {
111  arma::uword j = 0;
112  for (auto d=destination_begin; d != destination_end; ++d)
113  k_dists.at(i, j++) = scaled_dist(s->mid_point(), d->mid_point(), parameter);
114  ++i;
115  }
116  cov(k_dists, k, parameter);
117  }

272  template<class TSA, class S, class D, class T, class P>
273  void btk_interpolation(S source_begin, S source_end,
274  D destination_begin, D destination_end,
275  const T& time_axis, const P& parameter)
276  {
277  // Armadillo's notion of submatrix views are used for slicing out portions of F, k, and K
278  // to minimize the number of matrix allocations.
279 
280  // Allocate matrices of known sizes:
281  arma::mat22 H, H_inv, G, G_inv, GH_inv;
282  arma::mat::fixed<2,1> E_beta_pri, E_beta_w_pri, E_beta_post, beta_hat;
283  // These matrices sizes vary with the number valid sources and the number of destinations.
284  arma::mat K, k, F, f, K_inv, E_beta_w, omega, BM, T_obs, E_temp_post;
285 
286  // Prior data
287  E_beta_pri.at(0, 0) = 0.0; // Old code says this is ok. TODO: Check assumption.
288  arma::mat22 eye22 = arma::diagmat(arma::vec(2, arma::fill::ones));
289 
290  // Gather spatial data for all stations and destinations
291  utils::build_elevation_matrices(source_begin, source_end, destination_begin, destination_end, F, f);
292  utils::build_covariance_matrices(source_begin, source_end, destination_begin, destination_end, parameter, K, k);
293 
294  // Build full operators
295 
296  K_inv = K.i();
297  H_inv.operator=(F.t()*K_inv*F);
298  if (arma::rank(H_inv) == 1) {
299  throw std::runtime_error("The bayestian temperature kriging algorithm needs at least two sources at different heights.");
300  }
301  H.operator=(H_inv.i());
302  G_inv = H_inv;
303  G_inv.at(1, 1) += 1/(parameter.temperature_gradient_sd()*parameter.temperature_gradient_sd());
304  G.operator=(G_inv.i());
305  GH_inv.operator=(G*H_inv);
306  BM = (f - F.t()*K_inv*k).t()*(eye22 - GH_inv);
307  E_beta_w = H*F.t()*K_inv; // beta_est_weights
308  omega = k.t()*K_inv; // krig_weights
309  // Reduced matrices
310  arma::mat F_r, E_beta_w_r, omega_r, GH_inv_r, BM_r;
311 
312  // Matrix pointers used in the time loop
313  arma::mat *F_p=nullptr, *E_beta_w_p=nullptr, *omega_p=nullptr, *GH_inv_p=nullptr, *BM_p=nullptr;
314 
315  const size_t num_sources = std::distance(source_begin, source_end);
316  std::vector<TSA> source_accessors;
317  source_accessors.reserve(num_sources);
318  std::for_each(source_begin, source_end, [&] (const typename S::value_type& source)
319  { source_accessors.emplace_back(TSA(source.temperatures(), time_axis)); });
320  // Time step loop
321  const size_t num_timesteps = time_axis.size();
322  std::vector<double> temperatures;
323  temperatures.reserve(num_sources);
324  size_t idx;
325 
326  std::vector<arma::uword> valid_inds, prev_valid_inds;
327  valid_inds.reserve(num_sources);
328  for (size_t t_step=0; t_step < num_timesteps; ++t_step) {
329  temperatures.clear();
330  prev_valid_inds = valid_inds;
331  valid_inds.clear();
332  idx = 0;
333  for (auto& a: source_accessors) {
334  double v = a.value(t_step);
335  if (std::isfinite(v)) {
336  valid_inds.push_back((arma::uword)idx);
337  temperatures.push_back(v);
338  }
339  ++idx;
340  }
341  if (valid_inds != prev_valid_inds) {
342  if (valid_inds.size() == 0) {
343  std::cout << "period("<< t_step <<"| " << num_timesteps << ") = " << time_axis.period(t_step) << std::endl;
344  throw std::runtime_error("No valid sources for time period, giving up.");
345  }
346  if (valid_inds.size() == num_sources) {
347  // Use full operators
348  F_p = &F;
349  E_beta_w_p = &E_beta_w;
350  omega_p = &omega;
351  GH_inv_p = &GH_inv;
352  BM_p = &BM;
353  } else {
354  // Build new reduced operators from full operators
355  arma::uvec sub_idx(valid_inds);
356  F_r = F.rows(sub_idx);
357  K_inv = K.submat(sub_idx, sub_idx).i();
358  H_inv.operator=(F_r.t()*K_inv*F_r);
359  H.operator=(H_inv.i());
360  G_inv = H_inv;
361  G_inv.at(1, 1) += 1/(parameter.temperature_gradient_sd()*parameter.temperature_gradient_sd());
362  G.operator=(G_inv.i());
363  GH_inv_r = G*H_inv;
364  arma::mat k_red = k.rows(sub_idx);
365  BM_r = (f - F_r.t()*K_inv*k_red).t()*(eye22 - GH_inv_r);
366  E_beta_w_r = H*F_r.t()*K_inv; // beta_est_weights
367  omega_r = k_red.t()*K_inv; // krieg_weights
368  // Assign pointers
369  F_p = &F_r;
370  E_beta_w_p = &E_beta_w_r;
371  omega_p = &omega_r;
372  GH_inv_p = &GH_inv_r;
373  BM_p = &BM_r;
374  }
375  }
376 
377  // Build prior data for time step:
378  E_beta_pri.at(1, 0) = parameter.temperature_gradient(time_axis.period(t_step));
379  E_beta_w_pri.operator=((eye22 - *GH_inv_p)*E_beta_pri);
380 
381  // Fill T_obs with valid temperatures
382  T_obs.set_size((arma::uword)valid_inds.size(), 1);
383  std::copy(std::begin(temperatures), std::end(temperatures), T_obs.begin_col(0));
384  // Core computational work here:
385  beta_hat.operator=((*E_beta_w_p)*T_obs);
386  arma::mat T_hat = f.t()*beta_hat + (*omega_p)*(T_obs - (*F_p)*beta_hat);
387  E_beta_post.operator=((*GH_inv_p)*beta_hat + E_beta_w_pri);
388  E_temp_post = arma::vec(T_hat - (*BM_p)*(beta_hat - E_beta_pri));
389 
390  arma::uword dist = 0;
391  for (D d=destination_begin; d != destination_end; ++d)
392  d->set_temperature(t_step, E_temp_post.at(dist++));
393 
394  }
395  }
