769  size_t hint_based_search(const S& source, const utcperiod& p, size_t i) {
770  const size_t n = source.size();
771  if (n == 0)
772  return std::string::npos;
773  if (i != std::string::npos && i<n) { // hint-based search logic goes here:
774  const size_t max_directional_search = 5; // +-5 just a guess for what is a reasonable try upward/downward
775  auto ti = source.get(i).t;
776  if (ti == p.start) { // direct hit and extreme luck ?
777  return i; // just use it !
778  } else if (ti < p.start) { // do a local search upward to see if we find the spot we search for
779  if (i == n - 1) return i;// unless we are at the end (n-1), try to search upward
780  size_t i_max = std::min(i + max_directional_search, n);
781  while (++i < i_max) {
782  ti = source.get(i).t;
783  if (ti < p.start )
784  continue;
785  return ti > p.start? i - 1 : i;// we either got one to far, or direct-hit
786  }
787  return (i < n) ? source.index_of(p.start):n-1; // either local search failed->bsearch etc., or we are at the end -> n-1
788  } else if (ti > p.start) { // do a local search downwards from last index, maybe we are lucky
789  if (i == 0) // if we are at the beginning, just return npos (no left-bound index found)
790  return 0;//std::string::npos;
791  size_t i_min = (i - std::min(i, max_directional_search));
792  do {
793  ti = source.get(--i).t;//notice that i> 0 before we start due to if(i==0) above(needed due to unsigned i!)
794  if ( ti > p.start)
795  continue;
796  return i; // we found the lower left bound (either exact, or less p.start)
797  } while (i > i_min);
798  return i>0? source.index_of(p.start): std::string::npos; // i is >0, there is a hope to find the index using index_of, otherwise, no left lower bound
799  }
800  }
801  return source.index_of(p.start);// no hint given, just use binary search to establish the start.

822  template <class S>
823  double accumulate_value(const S& source, const utcperiod& p, size_t& last_idx, utctimespan& tsum,bool linear = true) {
824  const size_t n = source.size();
825  if (n == 0) // early exit if possible
826  return shyft::nan;
827  size_t i = hint_based_search(source, p, last_idx); // use last_idx as hint, allowing sequential periodic average to execute at high speed(no binary searches)
828 
829  if (i == std::string::npos) { // this might be a case
830  last_idx = 0;// we update the hint to the left-most possible index
831  return shyft::nan; // and is equivalent to no points, or all points after requested period.
832  }
833  point l;// Left point
834  bool l_finite = false;
835 
836  double area = 0.0; // Integrated area over the non-nan parts of the time-axis
837  tsum = 0; // length of non-nan f(t) time-axis
838 
839  while (true) { //there are two exit criteria: no more points, or we pass the period end.
840  if (!l_finite) {//search for 'point' anchor phase
841  l = source.get(i++);
842  l_finite = std::isfinite(l.v);
843  if (i == n) { // exit condition
844  if (l_finite && l.t < p.end) {//give contribution
845  utctimespan dt = p.end - l.t;
846  tsum += dt;
847  area += dt* l.v; // extrapolate value flat
848  }
849  break;//done
850  }
851  if (l.t >= p.end) {//also exit condition, if the point time is after period we search, then no anchor to be found
852  break;//done
853  }
854  } else { // got point anchor l, search for right point/end
855  point r = source.get(i++);// r is guaranteed > p.start due to hint-based search
856  bool r_finite = std::isfinite(r.v);
857  utcperiod px(std::max(l.t, p.start), std::min(r.t, p.end));
858  utctimespan dt = px.timespan();
859  tsum += dt;
860 
861  // now add area contribution for l..r
862  if (linear && r_finite) {
863  double a = (r.v - l.v) / (r.t - l.t);
864  double b = r.v - a*r.t;
865  //area += dt* 0.5*( a*px.start+b + a*px.end+b);
866  area += dt * (0.5*a*(px.start + px.end) + b);
867  } else { // flat contribution from l max(l.t,p.start) until max time of r.t|p.end
868  area += l.v*dt;
869  }
870  if (i == n) { // exit condition: final value in sequence, then we need to finish, but
871  if (r_finite && r.t < p.end) {// add area contribution from r and out to the end of p
872  dt = p.end - r.t;
873  tsum += dt;
874  area += dt* r.v; // extrapolate value flat
875  }
876  break;//done
877  }
878  if (r.t >= p.end)
879  break;//also exit condition, done
880  l_finite = r_finite;
881  l = r;
882  }
883 
884  }
885  last_idx = i - 1;
886  return tsum ? area : shyft::nan;
887  }
