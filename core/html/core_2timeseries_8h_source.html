108  size_t hint_based_search(const S& source, const utcperiod& p, size_t i) {
109  const size_t n = source.size();
110  if (n == 0)
111  return std::string::npos;
112  if (i != std::string::npos && i<n) { // hint-based search logic goes here:
113  const size_t max_directional_search = 5; // +-5 just a guess for what is a reasonable try upward/downward
114  auto ti = source.get(i).t;
115  if (ti == p.start) { // direct hit and extreme luck ?
116  return i; // just use it !
117  } else if (ti < p.start) { // do a local search upward to see if we find the spot we search for
118  if (i == n - 1) return i;// unless we are at the end (n-1), try to search upward
119  size_t i_max = std::min(i + max_directional_search, n);
120  while (++i < i_max) {
121  ti = source.get(i).t;
122  if (ti < p.start )
123  continue;
124  return ti > p.start? i - 1 : i;// we either got one to far, or direct-hit
125  }
126  return (i < n) ? source.index_of(p.start):n-1; // either local search failed->bsearch etc., or we are at the end -> n-1
127  } else if (ti > p.start) { // do a local search downwards from last index, maybe we are lucky
128  if (i == 0) // if we are at the beginning, just return npos (no left-bound index found)
129  return 0;//std::string::npos;
130  size_t i_min = (i - std::min(i, max_directional_search));
131  do {
132  ti = source.get(--i).t;//notice that i> 0 before we start due to if(i==0) above(needed due to unsigned i!)
133  if ( ti > p.start)
134  continue;
135  return i; // we found the lower left bound (either exact, or less p.start)
136  } while (i > i_min);
137  return i>0? source.index_of(p.start): std::string::npos; // i is >0, there is a hope to find the index using index_of, otherwise, no left lower bound
138  }
139  }
140  return source.index_of(p.start);// no hint given, just use binary search to establish the start.

164  double accumulate_value(const S& source, const utcperiod& p, size_t& last_idx, utctimespan& tsum,bool linear = true) {
165  const size_t n = source.size();
166  if (n == 0) // early exit if possible
167  return shyft::nan;
168  size_t i = hint_based_search(source, p, last_idx); // use last_idx as hint, allowing sequential periodic average to execute at high speed(no binary searches)
169 
170  if (i == std::string::npos) { // this might be a case
171  last_idx = 0;// we update the hint to the left-most possible index
172  return shyft::nan; // and is equivalent to no points, or all points after requested period.
173  }
174  point l;// Left point
175  bool l_finite = false;
176 
177  double area = 0.0; // Integrated area over the non-nan parts of the time-axis
178  tsum = 0; // length of non-nan f(t) time-axis
179 
180  while (true) { //there are two exit criteria: no more points, or we pass the period end.
181  if (!l_finite) {//search for 'point' anchor phase
182  l = source.get(i++);
183  l_finite = std::isfinite(l.v);
184  if (i == n) { // exit condition
185  if (l_finite && l.t < p.end) {//give contribution
186  utctimespan dt = p.end - l.t;
187  tsum += dt;
188  area += dt* l.v; // extrapolate value flat
189  }
190  break;//done
191  }
192  if (l.t >= p.end) {//also exit condition, if the point time is after period we search, then no anchor to be found
193  break;//done
194  }
195  } else { // got point anchor l, search for right point/end
196  point r = source.get(i++);// r is guaranteed > p.start due to hint-based search
197  bool r_finite = std::isfinite(r.v);
198  utcperiod px(std::max(l.t, p.start), std::min(r.t, p.end));
199  utctimespan dt = px.timespan();
200  tsum += dt;
201 
202  // now add area contribution for l..r
203  if (linear && r_finite) {
204  double a = (r.v - l.v) / (r.t - l.t);
205  double b = r.v - a*r.t;
206  //area += dt* 0.5*( a*px.start+b + a*px.end+b);
207  area += dt * (0.5*a*(px.start + px.end) + b);
208  } else { // flat contribution from l max(l.t,p.start) until max time of r.t|p.end
209  area += l.v*dt;
210  }
211  if (i == n) { // exit condition: final value in sequence, then we need to finish, but
212  if (r_finite && r.t < p.end) {// add area contribution from r and out to the end of p
213  dt = p.end - r.t;
214  tsum += dt;
215  area += dt* r.v; // extrapolate value flat
216  }
217  break;//done
218  }
219  if (r.t >= p.end)
220  break;//also exit condition, done
221  l_finite = r_finite;
222  l = r;
223  }
224 
225  }
226  last_idx = i - 1;
227  return tsum ? area : shyft::nan;
228  }
