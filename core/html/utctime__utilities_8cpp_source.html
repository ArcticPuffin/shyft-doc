235  utctime calendar::add(utctime t, utctimespan deltaT, long n) const {
236  auto dt=n*deltaT;
237  switch (deltaT) {
238  case YEAR: {
239  auto c=calendar_units(t);
240  c.year += int(dt/YEAR);// gives signed number of units.
241  return time(c);
242  } break;
243  case QUARTER://just let it appear as 3xmonth
244  deltaT = MONTH;
245  n = 3 * n;
246  // fall through to month
247  case MONTH:{
248  auto c=calendar_units(t);
249  // calculate number of years..
250  int nyears= int(dt/MONTH/12); // with correct sign
251  c.year += nyears; // done with years, now single step remaining months.
252  int nmonths= n-nyears*12;// remaining months to deal with
253  c.month += nmonths;// just add them
254  if(c.month <1) {c.month+=12; c.year--;}// then repair underflow
255  else if(c.month >12 ) {c.month -=12;c.year++;}// or overflow
256  return time(c);
257  } break;
258  default: break;
259  }
260  utctime r = t + dt; //naive first estimate
261  auto utc_diff_1=tz_info->utc_offset(t);
262  auto utc_diff_2=tz_info->utc_offset(r);
263  return r + (utc_diff_1-utc_diff_2);
264  // explanation: if t and r are in different dst, compensate for difference
265  // e.g.: t+ calendar day, in dst, will be 23, 24 or 25 hours long, this way
266  // 25: t= mm:dd 00:00, utcdiff(t) is +2h, add 24h
267  // r= mm:dd+1 00:00, utcdiff(r) is +1h
268  // ret r +( 2h - 1h), e.g. add one hour and we get 25h day.
269  // 24: trivial utcdiff before and after are same, 0.
270  //
271  // 23: t= mm:dd 00:00 utcdiff(t) is +1h, then we add 24h
272  // r= mm:dd+1 00:00 utcdiff(r) is +2h (since we are in summer time)
273  // ret r +( 1h - 2h), eg. sub one hour, and we get a 23h day.
274  };

276  utctimespan calendar::diff_units(utctime t1, utctime t2, utctimespan deltaT, utctimespan &remainder) const {
277  if (t1 == no_utctime || t2 == no_utctime || deltaT == 0L) {
278  remainder = 0L;
279  return 0L;
280  }
281  int sgn=1;
282  if(t1>t2) {sgn=-1;swap(t1,t2);}
283  utctimespan n_units= (t2-t1)/deltaT;// a *rough* estimate in case of dst or month/year
284  // recall that this should give n_unit=1, remainder=0 for dst days 23h | 25h etc.
285  // and that it should be complementary to the add(t,dt,n) function.
286  if (deltaT < DAY) {
287  if (deltaT > HOUR) {// a bit tricky, tz might jeopardize interval size by 1h (again assuming dst=1h)
288  auto utc_diff_1 = tz_info->utc_offset(t1);//assuming dst=1h!
289  auto utc_diff_2 = tz_info->utc_offset(t2);//assuming dst=1h!
290  n_units = (t2 - t1 - (utc_diff_1 - utc_diff_2)) / deltaT;
291  remainder = t2 - add(t1, deltaT, n_units);
292  } else { // simple case, hour< 15 min etc.
293  remainder = t2 - (t1 + n_units*deltaT);
294  }
295  } else {
296  if (deltaT == MONTH) {
297  n_units -= n_units / 72; // MONTH is 30 days, a real one is ~ 30.4375, after 72 months this adds up to ~ 1 month error
298  } else if (deltaT == QUARTER) {
299  n_units -= n_units /(3* 72);// 3x month compensation
300  } else if (deltaT == YEAR) {
301  n_units -= n_units / (4 * 365 * 365);// YEAR is 365, real is ~365.25, takes 5332900 years before it adds up to 1 year
302  }
303  auto t2x=add(t1,deltaT,long(n_units));
304  if(t2x>t2) { // got one step to far
305  --n_units;// reduce n_units, and calculate remainder
306  remainder= t2 - add(t1,deltaT,long(n_units));
307  } else if(t2x<t2) {// got one step short, there might be a hope for one more
308  ++n_units;// try adding one unit
309  auto t2xx= add(t1,deltaT,long(n_units));//calc new t2x
310  if(t2xx>t2) { // to much ?
311  --n_units;// reduce, and
312  remainder = t2 - t2x;// remainder is as with previous t2x
313  } else {// success, we could increase with one
314  remainder = t2 - t2xx;//new remainder based on t2xx
315  }
316  } else {
317  remainder=utctimespan(0);// perfect hit, zero remainder
318  }
319  }
320  return n_units*sgn;
321  };
