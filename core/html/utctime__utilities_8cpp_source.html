118  utctime calendar::trim(utctime t, utctimespan deltaT) const {
119  if (t == no_utctime || t == min_utctime || t == max_utctime || deltaT==utctimespan(0))
120  return t;
121  switch (deltaT) {
122  case YEAR: {
123  auto c = calendar_units(t);
124  c.month = c.day = 1; c.hour = c.minute = c.second = 0;
125  return time(c);
126  }break;
127  case MONTH: {
128  auto c = calendar_units(t);
129  c.day = 1; c.hour = c.minute = c.second = 0;
130  return time(c);
131  }break;
132  case DAY: {
133  auto c = calendar_units(t);
134  c.hour = c.minute = c.second = 0;
135  return time(c);
136  }break;
137  }
138  auto tz_offset=tz_info->utc_offset(t);
139  const utctime t0 = utctime(3LL * DAY) + utctime(WEEK * 52L * 2000LL);// + 3 days to get to a isoweek monday, then add 2000 years to get a positive number
140  t += t0 + tz_offset;
141  t= deltaT*(t / deltaT) - t0 ;
142  return t - tz_info->utc_offset(t);

145  utctime calendar::add(utctime t, utctimespan deltaT, long n) const {
146  auto dt=n*deltaT;
147  switch (deltaT) {
148  case YEAR: {
149  auto c=calendar_units(t);
150  c.year += int(dt/YEAR);// gives signed number of units.
151  return time(c);
152  } break;
153  case MONTH:{
154  auto c=calendar_units(t);
155  // calculate number of years..
156  int nyears= int(dt/MONTH/12); // with correct sign
157  c.year += nyears; // done with years, now single step remaining months.
158  int nmonths= n-nyears*12;// remaining months to deal with
159  c.month += nmonths;// just add them
160  if(c.month <1) {c.month+=12; c.year--;}// then repair underflow
161  else if(c.month >12 ) {c.month -=12;c.year++;}// or overflow
162  return time(c);
163  } break;
164  default: break;
165  }
166  utctime r= t+dt;
167  auto utc_diff_1=tz_info->utc_offset(t);
168  auto utc_diff_2=tz_info->utc_offset(r);
169  return r + (utc_diff_1-utc_diff_2);
170  // explanation: if t and r are in different dst, compensate for difference
171  // e.g.: t+ calendar day, in dst, will be 23, 24 or 25 hours long, this way
172  // 25: t= mm:dd 00:00, utcdiff(t) is +2h, add 24h
173  // r= mm:dd+1 00:00, utcdiff(r) is +1h
174  // ret r +( 2h - 1h), e.g. add one hour and we get 25h day.
175  // 24: trivial utcdiff before and after are same, 0.
176  //
177  // 23: t= mm:dd 00:00 utcdiff(t) is +1h, then we add 24h
178  // r= mm:dd+1 00:00 utcdiff(r) is +2h (since we are in summer time)
179  // ret r +( 1h - 2h), eg. sub one hour, and we get a 23h day.
180  };

182  utctimespan calendar::diff_units(utctime t1, utctime t2, utctimespan deltaT, utctimespan &remainder) const {
183  if (t1 == no_utctime || t2 == no_utctime || deltaT == 0L) {
184  remainder = 0L;
185  return 0L;
186  }
187  int sgn=1;
188  if(t1>t2) {sgn=-1;swap(t1,t2);}
189  utctimespan n_units= (t2-t1)/deltaT;// a *rough* estimate in case of dst or month/year
190  // recall that this should give n_unit=1, remainder=0 for dst days 23h | 25h etc.
191  // and that it should be complementary to the add(t,dt,n) function.
192  switch (deltaT) {
193  case calendar::MONTH: n_units -= n_units/72; // MONTH is 30 days, a real one is ~ 30.41, after 72 monts this adds up to ~ 1 month error
194  case calendar::YEAR:
195  case calendar::WEEK:
196  case calendar::DAY:{
197  auto t2x=add(t1,deltaT,long(n_units));
198  if(t2x>t2) { // got one step to far
199  --n_units;// reduce n_units, and calculate remainder
200  remainder= t2 - add(t1,deltaT,long(n_units));
201  } else if(t2x<t2) {// got one step short, there might be a hope for one more
202  ++n_units;// try adding one unit
203  auto t2xx= add(t1,deltaT,long(n_units));//calc new t2x
204  if(t2xx>t2) { // to much ?
205  --n_units;// reduce, and
206  remainder = t2 - t2x;// remainder is as with previous t2x
207  } else {// success, we could increase with one
208  remainder = t2 - t2xx;//new remainder based on t2xx
209  }
210  } else {
211  remainder=utctimespan(0);// perfect hit, zero remainder
212  }
213  } break;
214  default: {
215  if (deltaT > HOUR) {// a bit tricky, tz might jeopardize interval size by 1h (again assuming dst=1h)
216  auto utc_diff_1 = tz_info->utc_offset(t1);//assuming dst=1h!
217  auto utc_diff_2 = tz_info->utc_offset(t2);//assuming dst=1h!
218  n_units = (t2 - t1 - (utc_diff_1 - utc_diff_2)) / deltaT;
219  remainder = t2 - add(t1, deltaT, n_units);
220  } else { // simple case, hour< 15 min etc.
221  remainder = t2 - (t1 + n_units*deltaT);
222  }
223  } break;
224  }
225  return n_units*sgn;
226  };
