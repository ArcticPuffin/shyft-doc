201  const double kirchner_fraction = 1 - glacier_fraction;
202  const double cell_area_m2 = geo_cell_data.area();
203  const double glacier_area_m2 = geo_cell_data.area()*glacier_fraction;

209

233  + response.kirchner.q_avg * (kirchner_fraction-bare_lake_fraction);
234  response.charge_m3s =
235  + shyft::mmh_to_m3s(prec, cell_area_m2)
236  - shyft::mmh_to_m3s(response.ae.ae, cell_area_m2)
237  + response.gm_melt_m3s
238  - shyft::mmh_to_m3s(response.total_discharge, cell_area_m2);
239  // Possibly save the calculated values using the collector callbacks.
240  response_collector.collect(i, response);
241 
242  if(i+1==i_end)
243  state_collector.collect(i+1, state);
