182  auto wind_speed_accessor = ws_accessor_t(wind_speed, time_axis);
183 
184  R response;
185  const double total_lake_fraction = geo_cell_data.land_type_fractions_info().lake() + geo_cell_data.land_type_fractions_info().reservoir();
186  const double glacier_fraction = geo_cell_data.land_type_fractions_info().glacier();
187  const double kirchner_fraction = 1 - total_lake_fraction - glacier_fraction;
188  const double glacier_area_m2 = geo_cell_data.area()*glacier_fraction;
189  // Initialize the method stack
190  precipitation_correction::calculator p_corr(parameter.p_corr.scale_factor);
191  priestley_taylor::calculator pt(parameter.pt.albedo, parameter.pt.alpha);
192  skaugen::calculator<typename P::snow_parameter_t, typename S::snow_state_t, typename R::snow_response_t> skaugen_snow;
193  kirchner::calculator<kirchner::trapezoidal_average, typename P::kirchner_parameter_t> kirchner(parameter.kirchner);
194 
195  size_t i_begin = n_steps > 0 ? start_step : 0;
196  size_t i_end = n_steps > 0 ? start_step + n_steps : time_axis.size();
197  for (size_t i = i_begin; i < i_end; ++i) {

199  double temp = temp_accessor.value(i);
200  double rad = rad_accessor.value(i);
201  double rel_hum = rel_hum_accessor.value(i);

203  double wind_speed = wind_speed_accessor.value(i);
204  state_collector.collect(i, state);

211  period.timespan());
212  kirchner.step(period.start, period.end, state.kirchner.q, response.kirchner.q_avg, response.snow.outflow, response.ae.ae); //all units mm/h over 'same' area
213 
214  response.total_discharge = prec*total_lake_fraction
215  + m3s_to_mmh(response.gm_melt_m3s, geo_cell_data.area())
216  + response.kirchner.q_avg * kirchner_fraction;
