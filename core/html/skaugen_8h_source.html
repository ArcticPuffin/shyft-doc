55  static inline double sca_rel_red(unsigned long u, unsigned long n, double unit_size, double nu_a, double alpha) {
56  const double nu_m = ((double)u/n)*nu_a;
57  // Note; We use m (melt) in stead of s (smelt) due to language conversion
58  // Compute: Find X such that f_a(X) = f_m(X)
59  const gamma_dist g_m(nu_m, 1.0/alpha);
60  const gamma_dist g_a(nu_a, 1.0/alpha);
61  const double g_a_mean = boost::math::mean(g_a);
62  auto zero_func = [&] (const double& x) {
63  return boost::math::pdf(g_m, x) - boost::math::pdf(g_a, x); } ;
64 
65  double lower = boost::math::mean(g_m);
66  double upper = boost::math::tools::brent_find_minima(zero_func, 0.0, g_a_mean, 2).first; // TODO: Is this robust enough??
67  while (boost::math::pdf(g_m, lower) < boost::math::pdf(g_a, lower))
68  lower *= 0.9;
69 
70  boost::uintmax_t max_iter = 100;
71  boost::math::tools::eps_tolerance<double> tol(10); // 10 bit precition on result
72  typedef std::pair<double, double> result_t;
73  result_t res = boost::math::tools::bisect(zero_func, lower, upper, tol, max_iter);
74  //result_t res = boost::math::tools::toms748_solve(zero_func, lower, upper, tol, max_iter); // TODO: Boost bug causes this to crash!!
75  const double x = (res.first + res.second)*0.5; // TODO: Check that we converged
76  // Compute: {m,a} = \int_0^x f_{m,a} dx
77  const double m = boost::math::cdf(g_m, x);
78  const double a = boost::math::cdf(g_a, x);
79  return a + 1.0 - m;
80  }

345  static inline void compute_shape_vars(const statistics& stat,
346  unsigned long nnn,
347  unsigned long n,
348  unsigned long u,
349  double sca,
350  double rel_red_sca,
351  double& alpha,
352  double& nu) {
353  const double alpha_0 = stat.alpha_0;
354  const double nu_0 = stat.alpha_0*stat.unit_size;
355  const double dyn_var = nu/(alpha*alpha);
356  const double init_var = nu_0/(alpha_0*alpha_0);
357  double tot_var = 0.0;
358  double tot_mean = 0.0;
359  if (n > 0) { // Accumulation
360  if (nnn == 0) { // No previous snow pack
361  tot_var = n*init_var*(1 + (n - 1)*stat.c(n));
362  tot_mean = n*nu_0/alpha_0;
363  } else {
364  const double old_var_cov = (nnn + n)*init_var*(1 + ((nnn + n) - 1)*stat.c(nnn + n));
365  const double new_var_cov = n*init_var*(1 + (n - 1)*stat.c(n));
366  tot_var = old_var_cov*sca*sca + new_var_cov*(1.0 - sca)*(1.0 - sca);
367  tot_mean = (sca*(nnn + n) + (1.0 - sca)*n)*stat.unit_size;
368  }
369  }
370  if (u > 0) { // Ablation
371  const double factor = (dyn_var/(nnn*init_var) + 1.0 + (nnn - 1)*stat.c(nnn))/(2*nnn);
372  const double non_cond_mean = (nnn - u)*stat.unit_size;
373  tot_mean = non_cond_mean/(1.0 - rel_red_sca);
374  const unsigned long cond_u = lrint((1.0 - rel_red_sca)*nnn - (nnn - u));
375  const double auto_var = cond_u > 0 ? init_var*cond_u*(1.0 + (cond_u - 1.0)*stat.c(cond_u)) : 0.0;
376  const double cross_var = cond_u > 0 ? init_var*cond_u*2.0*factor*cond_u : 0.0;
377  tot_var = dyn_var + auto_var - cross_var;
378  }
379 
380  if (fabs(tot_mean) < 1.0e-7) {
381  nu = nu_0;
382  alpha = alpha_0;
383  return;
384  }
385  nu = tot_mean*tot_mean/tot_var;
386  alpha = nu/(stat.unit_size*lrint(tot_mean/stat.unit_size));
387  }
