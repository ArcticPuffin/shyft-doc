35  inline double integrate(vector<double> f, vector<double> x, size_t n, double a, double b, bool f_b_is_zero=false) {
36  size_t left = 0;
37  double area = 0.0;
38  double f_l = 0.0;
39  double x_l = a;
40  while (a > x[left]) ++left;
41 
42  if (fabs(a - x[left]) > 1.0e-8 && left > 0) { // Linear interpolation of start point
43  --left;
44  f_l = (f[left + 1] - f[left])/(x[left + 1] - x[left])*(a - x[left]) + f[left];
45  } else
46  f_l = f[left];
47 
48  while (left < n - 1) {
49  if (b >= x[left + 1]) {
50  area += 0.5*(f_l + f[left + 1])*(x[left + 1] - x_l);
51  x_l = x[left + 1];
52  f_l = f[left + 1];
53  ++left;
54  }
55  else {
56  if (! f_b_is_zero)
57  area += (f_l + 0.5*(f[left + 1] - f_l)/(x[left + 1] - x_l)*(b - x_l))*(b - x_l);
58  else
59  area += 0.5*f_l*(b - x_l);
60  break;
61  }
62  }
63  return area;
64  }

144  calculator(const P& p, S& state) {
145  // Simple trapezoidal rule to normalize the snow redistribution quartiles
146  auto s = p.s;
147  I = p.intervals;
148  I_n = I.size();
149 
150  const double mean = hbv_snow::integrate(s, I, I_n, I[0], I[I_n - 1]);
151  sd = s;
152  for (auto &sd_: sd) sd_ /= mean;
153 
154  double swe = state.swe;
155  double sca = state.sca;
156 
157  sp = vector<double>(I_n, 0.0);
158  sw = vector<double>(I_n, 0.0);
159  if (swe <= 1.0e-3 || sca <= 1.0e-3) {
160  state.swe = state.sca = 0.0;
161  } else {
162  for (size_t i = 0; i < I_n; ++i)
163  sp[i] = sca < I[i] ? 0.0 : s[i]*swe;
164 
165  swe = integrate(sp, I, I_n, 0.0, sca, true);
166 
167  if (swe < state.swe) {
168  const double corr1 = state.swe/swe*p.lw;
169  const double corr2 = state.swe/swe*(1.0 - p.lw);
170  for (size_t i = 0; i< I_n; ++i) {
171  sw[i] = corr1*sp[i];
172  sp[i] *= corr2;
173  }
174  } else
175  sw = vector<double>(I_n, 0.0);
176  }
177  }

179  void refreeze(double &sp, double &sw, const double rain, const double potmelt, const double lw) {
180  // Note that the above calculations might violate the mass balance due to rounding errors. A fix might be to
181  // replace sw by a sw_fraction, sp with s_tot, and compute sw and sp based on these.
182  if (sp > 0.0) {
183  if (sw + rain > -potmelt) {
184  sp -= potmelt;
185  sw += potmelt + rain;
186  if (sw > sp*lw) sw = sp*lw;
187  } else {
188  sp += sw + rain;
189  sw = 0.0;

217  template <class R> void step(S& s, R& r, shyft::core::utctime t0, shyft::core::utctime t1, const P& p, double prec, double temp) {
218  double swe = s.swe;
219  double sca = s.sca;
220  const double total_water = prec + swe;
221  double snow,rain;
222  if( temp < p.tx ) {snow=prec;rain= 0.0;}
223  else {snow= 0.0;rain=prec;}
224  double step_in_days = (t1 - t0)/86400.0;
225  swe += snow + sca*rain;
226  if (swe < 0.1) {
227  r.outflow = total_water;
228  fill(begin(sp), end(sp), 0.0);
229  fill(begin(sw), end(sw), 0.0);
230  s.swe = 0.0;
231  s.sca = 0.0;
232  return;
233  }
234  if (snow > 0.0) {
235  auto idx = sca_index(sca);
236  if (sca > 1.0e-5 && sca < 1.0 - 1.0e-5) {
237  if (idx == 0) {
238  sp[0] *= sca/(I[1] - I[0]);
239  sw[0] *= sca/(I[1] - I[0]);
240  } else {
241  sp[idx] *= (1.0 + (sca - I[idx])/(I[idx] - I[idx - 1]))/(1.0 + (I[idx + 1] - I[idx])/(I[idx] - I[idx - 1]));
242  sw[idx] *= (1.0 + (sca - I[idx])/(I[idx] - I[idx - 1]))/(1.0 + (I[idx + 1] - I[idx])/(I[idx] - I[idx - 1]));
243  }
244  }
245 
246  for (size_t i = 0; i < I_n; ++i) sp[i] += snow*sd[i];
247 
248  for (size_t i = I_n - 2; i > 0; --i)
249  if (sd[i] > 0.0) {
250  sca = sd[i + 1];
251  break;
252  } else
253  sca = sd[1];
254  }
255 
256  double potmelt = p.cx*step_in_days*(temp - p.ts);
257  const double lw = p.lw;
258 
259  if (potmelt < 0.0) {
260  potmelt *= p.cfr;
261  for (size_t i = 0; i < I_n; ++i)
262  refreeze(sp[i], sw[i], rain, potmelt, lw);
263  } else {
264  size_t idx = melt_index(potmelt);
265 
266  if (idx == 0) sca = 0.0;
267  else if (idx == I_n) sca = 1.0;
268  else {
269  if (sp[idx] > 0.0) sca = I[idx] - (I[idx] - I[idx - 1])*(potmelt - sp[idx])/(sp[idx - 1] = sp[idx]);
270  else sca = (1.0 - potmelt/sp[idx - 1])*(sca - I[idx - 1]) + I[idx - 1];
271  }
272 
273  for (size_t i = 0; i < I_n; ++i)
274  update_state(sp[i], sw[i], rain, potmelt, lw);
275  }
276 
277  if (sca < 1.0e-6) swe = 0.0;
278  else {
279  bool f_is_zero = sca >= 1.0 ? false : true;
280  swe = integrate(sp, I, I_n, 0, sca, f_is_zero);
281  swe += integrate(sw, I, I_n, 0, sca, f_is_zero);
282  }
283 
284  if (total_water < swe) {
285  if (total_water - swe < -1.0e-6) {// we do have about 15-16 digits accuracy
286  ostringstream buff;
287  buff << "Negative outflow: total_water (" << total_water << ") - swe (" << swe << ") = " << total_water - swe;
288  throw runtime_error(buff.str());
289  } else
290  swe = total_water;
291  }
292  r.outflow = total_water - swe;
293  s.swe = swe;
294  s.sca = sca;
295  }
296  };
