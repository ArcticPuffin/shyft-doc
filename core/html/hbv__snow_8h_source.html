33  inline double integrate(std::vector<double> f, std::vector<double> x, size_t n, double a, double b, bool f_b_is_zero=false) {
34  size_t left = 0;
35  double area = 0.0;
36  double f_l = 0.0;
37  double x_l = a;
38  while (a > x[left]) ++left;
39 
40  if (fabs(a - x[left]) > 1.0e-8 && left > 0) { // Linear interpolation of start point
41  --left;
42  f_l = (f[left + 1] - f[left])/(x[left + 1] - x[left])*(a - x[left]) + f[left];
43  } else
44  f_l = f[left];
45 
46  while (left < n - 1) {
47  if (b >= x[left + 1]) {
48  area += 0.5*(f_l + f[left + 1])*(x[left + 1] - x_l);
49  x_l = x[left + 1];
50  f_l = f[left + 1];
51  ++left;
52  }
53  else {
54  if (! f_b_is_zero)
55  area += (f_l + 0.5*(f[left + 1] - f_l)/(x[left + 1] - x_l)*(b - x_l))*(b - x_l);
56  else
57  area += 0.5*f_l*(b - x_l);
58  break;
59  }
60  }
61  return area;
62  }

132  calculator(const P& p, S& state) {
133  // Simple trapezoidal rule to normalize the snow redistribution quartiles
134  auto s = p.s;
135  I = p.intervals;
136  I_n = I.size();
137 
138  const double mean = hbv_snow::integrate(s, I, I_n, I[0], I[I_n - 1]);
139  sd = s;
140  for (auto &sd_: sd) sd_ /= mean;
141 
142  double swe = state.swe;
143  double sca = state.sca;
144 
145  sp = std::vector<double>(I_n, 0.0);
146  sw = std::vector<double>(I_n, 0.0);
147  if (swe <= 1.0e-3 || sca <= 1.0e-3) {
148  state.swe = state.sca = 0.0;
149  } else {
150  for (size_t i = 0; i < I_n; ++i)
151  sp[i] = sca < I[i] ? 0.0 : s[i]*swe;

158  for (size_t i = 0; i< I_n; ++i) {
159  sw[i] = corr1*sp[i];

161  }
162  } else
163  sw = std::vector<double>(I_n, 0.0);

207  double sca = s.sca;

209  const double snow = temp < p.tx ? prec : 0.0;

212  swe += snow + sca*rain;
213  if (swe < 0.1) {
214  r.outflow=total_water;
215  std::fill(begin(sp), end(sp), 0.0);
216  std::fill(begin(sw), end(sw), 0.0);
217  s.swe=0.0;
218  s.sca=0.0;
219  return;
220  }

223  if (sca > 1.0e-5 && sca < 1.0 - 1.0e-5) {
224  if (idx == 0) {

226  sw[0] *= sca/(I[1] - I[0]);
227  } else {
228  sp[idx] *= (1.0 + (sca - I[idx])/(I[idx] - I[idx - 1]))/(1.0 + (I[idx + 1] - I[idx])/(I[idx] - I[idx - 1]));
229  sw[idx] *= (1.0 + (sca - I[idx])/(I[idx] - I[idx - 1]))/(1.0 + (I[idx + 1] - I[idx])/(I[idx] - I[idx - 1]));
230  }
231  }
232 
233  for (size_t i = 0; i < I_n; ++i) sp[i] += snow*sd[i];
234 
235  for (size_t i = I_n - 2; i > 0; --i)
236  if (sd[i] > 0.0) {
237  sca = sd[i + 1];
238  break;
239  } else

241  }

245

258  }
