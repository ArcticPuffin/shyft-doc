36  inline double integrate(vector<double> f, vector<double> x, size_t n, double a, double b, bool f_b_is_zero=false) {
37  size_t left = 0;
38  double area = 0.0;
39  double f_l = 0.0;
40  double x_l = a;
41  while (a > x[left]) ++left;
42 
43  if (fabs(a - x[left]) > 1.0e-8 && left > 0) { // Linear interpolation of start point
44  --left;
45  f_l = (f[left + 1] - f[left])/(x[left + 1] - x[left])*(a - x[left]) + f[left];
46  } else
47  f_l = f[left];
48 
49  while (left < n - 1) {
50  if (b >= x[left + 1]) {
51  area += 0.5*(f_l + f[left + 1])*(x[left + 1] - x_l);
52  x_l = x[left + 1];
53  f_l = f[left + 1];
54  ++left;
55  }
56  else {
57  if (! f_b_is_zero)
58  area += (f_l + 0.5*(f[left + 1] - f_l)/(x[left + 1] - x_l)*(b - x_l))*(b - x_l);
59  else
60  area += 0.5*f_l*(b - x_l);
61  break;
62  }
63  }
64  return area;
65  }

156  sd = s;

164  if (swe <= 1.0e-3 || sca <= 1.0e-3) {
