284  auto rad_accessor = rad_accessor_t(rad, time_axis);
285 
286  // Initialize the method stack

288  priestley_taylor::calculator pt(parameter.pt.albedo, parameter.pt.alpha);
289  hbv_snow::calculator<typename P::snow_parameter_t, typename S::snow_state_t> snow(parameter.snow, state.snow);
290  hbv_soil::calculator<typename P::soil_parameter_t> soil(parameter.soil);
291  hbv_tank::calculator<typename P::tank_parameter_t> tank(parameter.tank);
292  R response;
293  const double total_lake_fraction = geo_cell_data.land_type_fractions_info().lake() + geo_cell_data.land_type_fractions_info().reservoir();
294  const double glacier_fraction = geo_cell_data.land_type_fractions_info().glacier();
295  const double land_fraction = 1 - glacier_fraction;
296  const double cell_area_m2 = geo_cell_data.area();
297  const double glacier_area_m2 = geo_cell_data.area()*glacier_fraction;//const double forest_fraction = geo_cell_data.land_type_fractions_info().forest();
298 
299  size_t i_begin = n_steps > 0 ? start_step : 0;
300  size_t i_end = n_steps > 0 ? start_step + n_steps : time_axis.size();
301  for (size_t i = i_begin; i < i_end; ++i) {
302  utcperiod period = time_axis.period(i);
303  double temp = temp_accessor.value(i);
304  double rad = rad_accessor.value(i);
305  double rel_hum = rel_hum_accessor.value(i);
306  double prec = p_corr.calc(prec_accessor.value(i));
307  state_collector.collect(i, state);
308

310

312  response.pt.pot_evapotranspiration = pt.potential_evapotranspiration(temp, rad, rel_hum)*calendar::HOUR; // mm/h
313  response.ae.ae = hbv_actual_evapotranspiration::calculate_step(
314  state.soil.sm, response.pt.pot_evapotranspiration,

321 
322  double bare_lake_fraction = total_lake_fraction*(1.0 - state.snow.sca);// only direct response on bare (no snow-cover) lakes
323  response.total_discharge =
324  std::max(0.0, prec - response.ae.ae)*bare_lake_fraction // when it rains, remove ae. from direct response
325  + m3s_to_mmh(response.gm_melt_m3s, cell_area_m2) // the glacier also direct
326  + response.tank.outflow * (land_fraction - bare_lake_fraction);// in summer, only dry land response, during winter, let precip/snow go through tank
327  response.charge_m3s =
328  + shyft::mmh_to_m3s(prec, cell_area_m2)
329  - shyft::mmh_to_m3s(response.ae.ae, cell_area_m2)
330  + response.gm_melt_m3s
331  - shyft::mmh_to_m3s(response.total_discharge, cell_area_m2);
332  // Possibly save the calculated values using the collector callbacks.
333  response_collector.collect(i, response);
334  if (i + 1 == i_end)
335  state_collector.collect(i + 1, state);
336  }
337  response_collector.set_end_response(response);
338  }
