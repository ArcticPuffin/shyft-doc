323 double dream::std_norm() const {
324  double u1, u2, tmp;
325  if (super_hack_stored) {
326  // An extra random number was generated the last time std_norm() was called, simply return it and delete it.
327  super_hack_stored = false;
328  return stored_std_norm_super_hack;
329  } else {
330  // Draw two standard normal random numbers.
331  // rng_unif11() returns a uniform random between -1 and 1, non-inclusive, and is better than
332  // the system-supplied rand() due to utilization of 32-bit maxima (opposed to 16-bit).
333  // The polar-coordinate transformation to Normal avoids the time-consuming calls to trigonometric
334  // functions more frequently employed.
335  // Routines from Numerical Recepies in C.
336  do {
337  u1 = random11();
338  u2 = random11();
339  tmp = u1 * u1 + u2 * u2;
340  } while (tmp >= 1.0 || tmp == 0.0);
341  }
342  tmp = sqrt(-2.0 * log(tmp) / tmp);
343  stored_std_norm_super_hack = u1 * tmp;
344  super_hack_stored=true;
345  return u2 * tmp;
346 }

354 void dream::generate_candidate_parameters(vector<double> &x_candidate, // Returned vector of length d
355  const size_t I, // Current chain number
356  const size_t N, // Number of chains
357  const size_t d, // Number of parameters
358  const double cr, // cross-over probability
359  size_t &d_eff, // # actually changed params
360  const vector<vector<double>>& states) // Current states (parameter values) for all chains
361 const {
362  size_t delta = 1 + (size_t)floor(3 * random01()); // 1, 2, or 3 pairs used to generate a proposal
363  delta = min(delta, (N - 1) / 2); // If N<3, delta is 0, since it is not possible
364  delta = min(delta, (size_t)3); // to generate a pair of chains other than I.
365  size_t R[6]; // In that case, only b^* will affect proposal deviation.
366  for (size_t k = 0; k < 2 * delta; ++k) { // With delta equal to 1, 2, or 3,
367  R[k] = I; // draw two, four or six different R values
368  while (R[k] == I) { // less than N and different from I.
369  R[k] = min(N - 1, (size_t) floor(random01() * N));
370  for (size_t j = 0; j < k; ++j) // Check formerly drawn values of R
371  if (R[k] == R[j]) // Value is used before,
372  R[k] = I; // set to I to ensure redraw.
373  }
374  }
375 
376  for (size_t i = 0; i < d; ++i)
377  x_candidate[i] = 0;
378 
379  // Generate a candidate point for chain i by adding up k difference-vectors
380  // between randomly selected pairs of the other chains.
381  for (size_t k = 0; k < delta; ++k) {
382  for (size_t i = 0; i < d; ++i)
383  x_candidate[i] += states[R[2 * k]][i] // For each dim, accumulate the pair diff over delta pairs.
384  - states[R[2 * k + 1]][i]; // If the chains are wide apart, the proposed jumps will
385  } // tend to be larger than when the chains agree.
386 
387  // Replace each element j=1,...,d in the proposal with the current coordinate (parameter value) with a cross-over probability cr.
388  d_eff = d; // Number of dimensions in which candidate differs from current
389  for (size_t i = 0; i < d; ++i) {
390  if (random01() >= cr) {
391  // Keep the current value for this parameter
392  --d_eff;// d_eff holds the number of altered parameters where the candidate differs from the current state #i.
393  x_candidate[i] = -numeric_limits<double>::max();
394  }
395  }
396 
397  double gamma; // Depends on num dims, num altered dims, num pairs
398  if (random01() < 0.2)
399  gamma = 1; // Facilitate mode jumping in 20% of the generations.
400  else
401  gamma = 2.38 / sqrt(2.0 * delta * d_eff); // If d_eff==0, gamma is undefined, but is never used, since x_candidate[j]will be -9999 for all j.
402 
403  double b = 0.05; // Vrugt et al page 278, Case Studies.
404  double sd = 0.001; // sd is the square root of Vrugt et al's b^*.
405 
406  for (size_t i = 0; i < d; ++i) {
407  if (x_candidate[i] == -numeric_limits<double>::max()) // The j'th dimension was selected to ignore update proposal
408  x_candidate[i] = states[I][i]; // Candpoint stays at current value for this parameter
409  else { // Candpoint adopts the proposed value for this parameter
410  double e = (double) random11() * b; // Randomness in the increment direction
411  x_candidate[i] = states[I][i]
412  + (1 + e) * gamma * x_candidate[i]
413  + (double) normal(0, sd); // Additive random vector component, epsilon in Vrugt et al.
414  }
415  }
416 }

436 bool dream::check_for_outlier_chain(const vector<vector<double>>& chain_probabilities, size_t reset, vector<double>& omega, double &lPlim) const {
437  const size_t n_iterations=chain_probabilities.size();
438  const size_t n_chains=omega.size();
439  size_t start = reset + (n_iterations - reset) / 2; //The first iteration used in calculating omega
440  if (start > n_iterations) return false;
441  size_t length = n_iterations - start; // Number of iterations used in calculating omega (last 50% of the period since last replacement)
442  if (length < 5) return false;
443 
444  // Initialize the omega[] array
445  fill(begin(omega),end(omega),0.0);
446  // Calculate the omega array by first summing up log posterior likelihood (LPD)
447  // for each chain, and over the last [length] iterations.
448  for (size_t i = start; i < n_iterations; ++i)
449  for (size_t j = 0; j < n_chains; ++j)
450  omega[j] += chain_probabilities[i][j];
451  for (size_t i = 0; i < n_chains; ++i)
452  omega[i] /= (length); // Average posterior log-likelihood for each chain
453 
454  // The array omega[] is the unsorted chain ordered table.
455  // Now we must create array sorted[] should contain the same values but sorted.
456  auto sorted(omega);
457  sort(begin(sorted),end(sorted));
458 
459  // Calculate the LPD limit value to be used to decide from the omega values if a chain is an outlier
460  lPlim = sorted[n_chains / 4] // The 25% quantile Q1
461  - 2 *
462  (sorted[3 * n_chains / 4] // The 75% quantile Q3
463  - sorted[n_chains / 4]); // The 25% quantile Q1; Q3-Q1=IQR
464 
465  // Search all chains and check if at least one of them has an LPD lower than the limit
466  for (size_t i = 0; i < n_chains; ++i)
467  if (omega[i] < lPlim)
468  return true; // Chain i has been detected as an outlier, so we have at least one outlier chain!
469 
470  // If not returned, then we have no outlier chains.
471  return false;

480 double dream::get_gr_convergence(const vector<vector<vector<double>>>&/* double*** */ states, size_t n_iterations, size_t n_chains, size_t n_parameters, size_t reset, size_t parameter) const {
481  // Argument states contains the chain states for all iterations as a 3D array: Iterations, chain and parameter.
482  // Arguments n_iterations, n_chains and n_parameters gives us the size of the three dimensions.
483  // Argument reset is index of last outlier-reset.
484  // Argument parameter is the index of the current parameter, referring to the last dimension of input array states, to calculate GRC for.
485 
486  if (n_iterations < 1 || n_chains < 1 || n_parameters < 1) return -1.0;
487  size_t start = reset + (n_iterations - reset) / 2; //The first iteration used in calculating omega
488  if (start > n_iterations) return -1.0; // Return dummy GRC if too few data to estimate
489  size_t length = n_iterations - start; // Number of iterations used in calculating omega
490  if (length < 5) return -1.0; // Return dummy GRC if too few data to estimate
491 
492  double* chainmean = __autoalloc__(double, n_chains);
493  double* chainvar = __autoalloc__(double, n_chains);
494 
495  // Follow Gelman & Rubin's symbolism, and convert from int to double to be used in calculations
496  double m = (double) n_chains;
497  double n = (double) length;
498 
499  double meanchainmean = 0; // Mean of chain averages
500  double meanchainvar = 0; // Mean of chain variances
501  double varchainmean = 0; // Variance of chain averages
502  double varchainvar = 0; // Variance of chain variances
503  double meansqchainmean = 0; // Mean of squared chain averages
504 
505  for (size_t i = 0; i < n_chains; ++i) {
506  chainmean[i] = chainvar[i] = 0.0;
507  for (size_t j = start; j < n_iterations; ++j) {
508  double v = states[j][i][parameter];
509  chainmean[i] += v;
510  chainvar[i] += pow(v, 2);
511  }
