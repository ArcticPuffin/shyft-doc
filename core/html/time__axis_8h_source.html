578  template<class TA, class TB>
579  inline period_list combine( const TA& a, const TB & b, typename enable_if < !TA::continuous::value || !TB::continuous::value >::type* x = 0 ) {
580  utcperiod pa = a.total_period();
581  utcperiod pb = b.total_period();
582  if( !pa.overlaps( pb ) || a.size() == 0 || b.size() == 0 )
583  return period_list::null_range();
584  if( pa == pb && a.size() == b.size() ) { //possibly exact equal ?
585  bool all_equal = true;
586  for( size_t i = 0; i < a.size(); ++i ) {
587  if( a.period( i ) != b.period( i ) ) {
588  all_equal = false; break;
589  }
590  }
591  if( all_equal )
592  return period_list::convert( a );
593  }
594 
595  // the hard way merge points in the intersection of periods
596  utctime t0 = std::max( pa.start, pb.start );
597  utctime te = std::min( pa.end, pb.end );
598  utcperiod tp( t0, te );
599  size_t ia = a.open_range_index_of( t0 ); //notice these have to be non-nan!
600  size_t ib = b.open_range_index_of( t0 );
601  if( ia == string::npos || ib == string::npos )
602  throw runtime_error( "period_list::combine algorithmic error npos not expected here" );
603  size_t ea = 1 + a.open_range_index_of( te );
604  size_t eb = 1 + b.open_range_index_of( te );
605  period_list r;
606  r.p.reserve( ( ea - ia ) + ( eb - ib ) ); //assume worst case here, avoid realloc
607 
608  while( ia < ea && ib < eb ) { // while both do have contributions
609  utcperiod p_ia = intersection( a.period( ia ), tp );
610  utcperiod p_ib = intersection( b.period( ib ), tp );
611  if( p_ia.timespan() == 0 ) {++ia; continue;} // no contribution from a, skip to next a
612  if( p_ib.timespan() == 0 ) {++ib; continue;} // no contribution from b, skip to next b
613  utcperiod p_i = intersection( p_ia, p_ib ); // compute the intersection
614  if( p_i.timespan() == 0 ) { // no overlap|intersection
615  if( p_ia.start < p_ib.start ) ++ia; //advance the left-most interval
616  else if( p_ib.start < p_ia.start ) ++ib;
617  else {++ia; ++ib;} //TODO: should not be possible ? since it's not overlapping start cant be equal(except one is empty period)
618  continue;
619  }
620 
621  if( p_ia == p_ib ) {
622  r.p.push_back( p_ia ); ++ia; ++ib; // common period, push&incr. both
623  } else if( p_ib.contains( p_ia ) ) {
624  r.p.push_back( p_ia ); ++ia; // a contribution only, incr. a
625  } else if( p_ia.contains( p_ib ) ) {
626  r.p.push_back( p_ib ); ++ib; // b contribution only, incr. b
627  } else { //ok, not equal, and neither a or b contains the other
628  if( p_ia.start < p_ib.start ) {
629  r.p.push_back( p_i ); ++ia; // a contribution only, incr. a
630  } if( p_ib.start < p_ia.start ) {
631  r.p.push_back( p_i ); ++ib; // b contribution only, incr. b
632  }
633  }
634  }
635  return r;
636  }
