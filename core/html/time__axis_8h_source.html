580  utcperiod pb = b.total_period();
581  if( !pa.overlaps( pb ) || a.size() == 0 || b.size() == 0 )
582  return period_list::null_range();
583  if( pa == pb && a.size() == b.size() ) { //possibly exact equal ?
584  bool all_equal = true;
585  for( size_t i = 0; i < a.size(); ++i ) {
586  if( a.period( i ) != b.period( i ) ) {
587  all_equal = false; break;
588  }
589  }
590  if( all_equal )
591  return period_list::convert( a );
592  }
593 
594  // the hard way merge points in the intersection of periods
595  utctime t0 = std::max( pa.start, pb.start );
596  utctime te = std::min( pa.end, pb.end );
597  utcperiod tp( t0, te );
598  size_t ia = a.open_range_index_of( t0 ); //notice these have to be non-nan!
599  size_t ib = b.open_range_index_of( t0 );
600  if( ia == string::npos || ib == string::npos )
601  throw runtime_error( "period_list::combine algorithmic error npos not expected here" );
602  size_t ea = 1 + a.open_range_index_of( te );
603  size_t eb = 1 + b.open_range_index_of( te );
604  period_list r;
605  r.p.reserve( ( ea - ia ) + ( eb - ib ) ); //assume worst case here, avoid realloc
606 
607  while( ia < ea && ib < eb ) { // while both do have contributions
608  utcperiod p_ia = intersection( a.period( ia ), tp );
609  utcperiod p_ib = intersection( b.period( ib ), tp );
610  if( p_ia.timespan() == 0 ) {++ia; continue;} // no contribution from a, skip to next a
611  if( p_ib.timespan() == 0 ) {++ib; continue;} // no contribution from b, skip to next b
612  utcperiod p_i = intersection( p_ia, p_ib ); // compute the intersection
613  if( p_i.timespan() == 0 ) { // no overlap|intersection
614  if( p_ia.start < p_ib.start ) ++ia; //advance the left-most interval
615  else if( p_ib.start < p_ia.start ) ++ib;
616  else {++ia; ++ib;} //TODO: should not be possible ? since it's not overlapping start cant be equal(except one is empty period)
617  continue;
618  }
619 
620  if( p_ia == p_ib ) {
621  r.p.push_back( p_ia ); ++ia; ++ib; // common period, push&incr. both
622  } else if( p_ib.contains( p_ia ) ) {
623  r.p.push_back( p_ia ); ++ia; // a contribution only, incr. a
624  } else if( p_ia.contains( p_ib ) ) {
625  r.p.push_back( p_ib ); ++ib; // b contribution only, incr. b
626  } else { //ok, not equal, and neither a or b contains the other
627  if( p_ia.start < p_ib.start ) {
628  r.p.push_back( p_i ); ++ia; // a contribution only, incr. a
629  } if( p_ib.start < p_ia.start ) {
630  r.p.push_back( p_i ); ++ib; // b contribution only, incr. b
631  }
632  }
633  }
634  return r;
635  }

643  template <typename T_A, typename T_B, typename C = void >
644  struct combine_type { // generic fallback to period_list type, very general, but expensive
645  typedef period_list type;
646  };

650  struct combine_type<fixed_dt, fixed_dt, void> {typedef fixed_dt type;};
