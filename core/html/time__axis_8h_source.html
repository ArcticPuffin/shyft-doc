682  return ix_hint;// this is the closest period <= to tx

685  }

1286  size_t eb = 1 + b.open_range_index_of( te );

1291  utcperiod p_ia = intersection( a.period( ia ), tp );

1293  if( p_ia.timespan() == 0 ) {++ia; continue;} // no contribution from a, skip to next a
1294  if( p_ib.timespan() == 0 ) {++ib; continue;} // no contribution from b, skip to next b
1295  utcperiod p_i = intersection( p_ia, p_ib ); // compute the intersection
1296  if( p_i.timespan() == 0 ) { // no overlap|intersection
1297  if( p_ia.start < p_ib.start ) ++ia; //advance the left-most interval
1298  else if( p_ib.start < p_ia.start ) ++ib;
1299  else {++ia; ++ib;} //TODO: should not be possible ? since it's not overlapping start cant be equal(except one is empty period)

1306  r.p.push_back( p_ia ); ++ia; // a contribution only, incr. a
1307  } else if( p_ia.contains( p_ib ) ) {
