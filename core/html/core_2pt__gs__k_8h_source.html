296  auto wind_speed_accessor = wind_speed_accessor_t(wind_speed, time_axis);

300  // Initialize the method stack

302  priestley_taylor::calculator pt(parameter.pt.albedo, parameter.pt.alpha);
303  gamma_snow::calculator<typename P::gs_parameter_t, typename S::gs_state_t, typename R::gs_response_t> gs;
304  kirchner::calculator<kirchner::trapezoidal_average, typename P::kirchner_parameter_t> kirchner(parameter.kirchner);
305  //
306  // Get the initial states
307  R response;
308  const double forest_fraction=geo_cell_data.land_type_fractions_info().forest();
309  const double total_lake_fraction = geo_cell_data.land_type_fractions_info().lake() + geo_cell_data.land_type_fractions_info().reservoir(); // both give direct response for now
310  const double glacier_fraction = geo_cell_data.land_type_fractions_info().glacier();
311  const double kirchner_fraction = 1 - glacier_fraction;
312  const double cell_area_m2 = geo_cell_data.area();
313  const double glacier_area_m2 = geo_cell_data.area()*glacier_fraction;
314  const double altitude= geo_cell_data.mid_point().z;
315  // Step through times in axis
316  size_t i_begin = n_steps > 0 ? start_step : 0;
317  size_t i_end = n_steps > 0 ? start_step + n_steps : time_axis.size();
318  for (size_t i = i_begin ; i < i_end ; ++i) {
319  utcperiod period = time_axis.period(i);
320  double temp = temp_accessor.value(i);
321  double rad = rad_accessor.value(i);
322  double rel_hum = rel_hum_accessor.value(i);
323  double prec = p_corr.calc(prec_accessor.value(i));
324  state_collector.collect(i, state);
325 
326  gs.step(state.gs, response.gs, period.start, period.timespan(), parameter.gs,
327  temp, rad, prec, wind_speed_accessor.value(i), rel_hum,forest_fraction,altitude);
328  response.gm_melt_m3s = glacier_melt::step(parameter.gm.dtf, temp, geo_cell_data.area()*response.gs.sca, glacier_area_m2);
329  response.pt.pot_evapotranspiration = pt.potential_evapotranspiration(temp, rad, rel_hum)*calendar::HOUR; //mm/s -> mm/h
330  response.ae.ae = actual_evapotranspiration::calculate_step(state.kirchner.q, response.pt.pot_evapotranspiration,
331  parameter.ae.ae_scale_factor, std::max(response.gs.sca,glacier_fraction), // a evap only on non-snow/non-glac area
332  period.timespan());

334 
335  double bare_lake_fraction = total_lake_fraction*(1.0 - response.gs.sca);// only direct response on bare (no snow-cover) lakes
336  response.total_discharge =
337  std::max(0.0,prec - response.ae.ae)*bare_lake_fraction // when it rains, remove ae. from direct response
338  + shyft::m3s_to_mmh(response.gm_melt_m3s,cell_area_m2)
339  + response.kirchner.q_avg*(kirchner_fraction-bare_lake_fraction);//let kirchner respond to all except glacier and direct lake
340  response.charge_m3s =
341  + shyft::mmh_to_m3s(prec, cell_area_m2)
342  - shyft::mmh_to_m3s(response.ae.ae, cell_area_m2)
343  + response.gm_melt_m3s
344  - shyft::mmh_to_m3s(response.total_discharge, cell_area_m2);
345  // Possibly save the calculated values using the collector callbacks.
346  response_collector.collect(i, response);
347  if(i+1==i_end)
348  state_collector.collect(i+1, state);
349  }
350  response_collector.set_end_response(response);
